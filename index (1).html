Security
How it keeps your data safe
🖥

Uses the Windows Shell API — not a hack

The app calls the official Windows SysListView32 control to show or hide desktop icons. This is the same mechanism used by Windows itself — no low-level tricks, no memory patching, no driver installation.

📦

Your files are never touched

Hiding icons does not move, delete, encrypt, or modify a single file. It simply tells Windows to stop rendering the icon layer on your desktop. Toggle it back and everything reappears exactly as you left it.

🌐

No network access — ever

The app has no internet connectivity. It writes one small config file to your home folder (~/.desktop_icon_toggle/config.json) to remember your last state. That is the only thing it writes. Nothing is sent anywhere.

🔬

Fully auditable — open source, MIT licensed

Every line of code is on GitHub. You can read it, compile it yourself, or run it directly from source with Python. There is no obfuscation, no telemetry, and no hidden behaviour.

"""
Desktop Icon Toggle - System Tray App
======================================
Hides/shows desktop icons on double-click.
Runs silently in the system tray.

Requirements:
    pip install pystray pillow pywin32

Run:
    pythonw desktop_icon_toggle.py    (no console window)
    python desktop_icon_toggle.py     (with console, for testing)
"""

import ctypes
import threading
import sys
import json
import os
from pathlib import Path
from PIL import Image, ImageDraw
import pystray
import psutil

# Windows API constants
WM_COMMAND = 0x0111
MOD_ALT    = 0x0001
VK_D       = 0x44

user32  = ctypes.windll.user32
shell32 = ctypes.windll.shell32

# Configuration file path
CONFIG_FILE = Path.home() / ".desktop_icon_toggle" / "config.json"

# Global state
icons_visible = True


def check_single_instance():
    """Ensure only one instance runs. Kill others if found."""
    current_pid = os.getpid()
    script_name = "desktop_icon_toggle.py"
    for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
        try:
            cmdline = proc.info['cmdline']
            if cmdline and script_name in ' '.join(cmdline):
                if proc.info['pid'] != current_pid:
                    proc.kill()
        except (psutil.NoSuchProcess, psutil.AccessDenied):
            pass


def load_config():
    global icons_visible
    if CONFIG_FILE.exists():
        with open(CONFIG_FILE, 'r') as f:
            config = json.load(f)
            icons_visible = config.get('icons_visible', True)


def save_config():
    CONFIG_FILE.parent.mkdir(parents=True, exist_ok=True)
    with open(CONFIG_FILE, 'w') as f:
        json.dump({'icons_visible': icons_visible}, f)


def get_desktop_hwnd():
    """Find the ShellDefView window that controls desktop icons."""
    progman = user32.FindWindowW("Progman", None)
    user32.SendMessageTimeoutW(progman, 0x052C, 0, 0, 0x0000, 1000, None)
    desktop_hwnd = [None]

    def enum_windows_proc(hwnd, lParam):
        shell_dll_view = user32.FindWindowExW(hwnd, None, "SHELLDLL_DefView", None)
        if shell_dll_view:
            desktop_hwnd[0] = shell_dll_view
        return True

    EnumProc = ctypes.WINFUNCTYPE(ctypes.c_bool, ctypes.POINTER(ctypes.c_int), ctypes.POINTER(ctypes.c_int))
    user32.EnumWindows(EnumProc(enum_windows_proc), None)

    if not desktop_hwnd[0]:
        desktop_hwnd[0] = user32.FindWindowExW(progman, None, "SHELLDLL_DefView", None)
    return desktop_hwnd[0]


def set_desktop_icons_visible(visible: bool):
    shell_view = get_desktop_hwnd()
    if not shell_view:
        import winreg
        key_path = r"Software\Microsoft\Windows\CurrentVersion\Explorer\Advanced"
        with winreg.OpenKey(winreg.HKEY_CURRENT_USER, key_path, 0, winreg.KEY_SET_VALUE) as key:
            winreg.SetValueEx(key, "HideIcons", 0, winreg.REG_DWORD, 0 if visible else 1)
        shell32.SHChangeNotify(0x8000000, 0x1000, None, None)
        return
    show_flag = 5 if visible else 0
    list_view = user32.FindWindowExW(shell_view, None, "SysListView32", None)
    if list_view:
        user32.ShowWindow(list_view, show_flag)


def create_tray_icon_image(visible: bool) -> Image.Image:
    size = 64
    img  = Image.new("RGBA", (size, size), (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)
    if visible:
        draw.ellipse([4, 4, 60, 60], fill=(40, 167, 69, 255))
        for row in range(2):
            for col in range(2):
                x = 16 + col * 18
                y = 16 + row * 18
                draw.rectangle([x, y, x+12, y+12], fill=(255, 255, 255, 200))
    else:
        draw.ellipse([4, 4, 60, 60], fill=(108, 117, 125, 255))
        draw.line([20, 20, 44, 44], fill=(255, 255, 255, 220), width=5)
        draw.line([44, 20, 20, 44], fill=(255, 255, 255, 220), width=5)
    return img


def toggle_icons(icon, item=None):
    global icons_visible
    icons_visible = not icons_visible
    set_desktop_icons_visible(icons_visible)
    save_config()
    update_tray(icon)


def update_tray(icon):
    icon.icon  = create_tray_icon_image(icons_visible)
    icon.title = ("Desktop Icons: VISIBLE\nDouble-click to hide"
                  if icons_visible
                  else "Desktop Icons: HIDDEN\nDouble-click to show")


def quit_app(icon, item):
    if not icons_visible:
        set_desktop_icons_visible(True)
    icon.stop()


def setup_autostart(enable: bool):
    import winreg
    key_path = r"Software\Microsoft\Windows\CurrentVersion\Run"
    app_name = "DesktopIconToggle"
    exe_path = f'pythonw "{sys.argv[0]}"'
    with winreg.OpenKey(winreg.HKEY_CURRENT_USER, key_path, 0, winreg.KEY_SET_VALUE) as key:
        if enable:
            winreg.SetValueEx(key, app_name, 0, winreg.REG_SZ, exe_path)
        else:
            try: winreg.DeleteValue(key, app_name)
            except FileNotFoundError: pass


def main():
    global icons_visible
    check_single_instance()
    load_config()
    setup_autostart(True)
    set_desktop_icons_visible(icons_visible)

    image = create_tray_icon_image(icons_visible)
    menu  = pystray.Menu(
        pystray.MenuItem("Show / Hide Icons", toggle_icons, default=True),
        pystray.Menu.SEPARATOR,
        pystray.MenuItem("Disable autostart on login", lambda icon, item: setup_autostart(False)),
        pystray.Menu.SEPARATOR,
        pystray.MenuItem("Quit (restores icons)", quit_app),
    )
    icon = pystray.Icon(
        name="DesktopIconToggle",
        icon=image,
        title="Desktop Icons: HIDDEN\nDouble-click to show" if not icons_visible
              else "Desktop Icons: VISIBLE\nDouble-click to hide",
        menu=menu,
    )
    icon.on_double_click = on_double_click
    icon.run()

if __name__ == "__main__":
    main()

$WshShell = New-Object -ComObject WScript.Shell; ^

$Shortcut.TargetPath = 'pythonw'; ^
$Shortcut.Arguments = '%SCRIPT_PATH%'; ^
$Shortcut.WorkingDirectory = '%SCRIPT_DIR%'; ^
$Shortcut.WindowStyle = 7; ^
$Shortcut.IconLocation = 'shell32.dll, 44'; ^
$Shortcut.Save(); ^
Write-Host 'Shortcut created: ' $ShortcutPath"

echo.
echo Setup complete! The app will start automatically on login.
